   101→function SortableColumnItem({ column, onRemove }: { column: { key: string; label: string }; onRemove: (key: string) => void }) {
   102→  const { attributes, listeners, setNodeRef, transform, transition } = useSortable({ id: column.key });
   103→  const style = { transform: CSS.Transform.toString(transform), transition };
   104→  return (
   105→    <div ref={setNodeRef} style={style} className="flex items-center gap-2 p-2 bg-slate-50 border border-slate-200 rounded-md hover:bg-slate-100 transition-colors">
   106→      <button {...attributes} {...listeners} className="cursor-grab active:cursor-grabbing text-slate-400 hover:text-slate-600 shrink-0">
   107→        <GripVertical className="h-4 w-4" />
   108→      </button>
   109→      <span className="flex-1 text-sm text-slate-700 truncate">{column.label}</span>
   110→      <Button variant="ghost" size="sm" className="h-6 w-6 p-0 text-slate-400 hover:text-red-500 hover:bg-red-50 shrink-0" onClick={() => onRemove(column.key)}>
   111→        <X className="h-3.5 w-3.5" />
   112→      </Button>
   113→    </div>
   114→  );
   115→}
